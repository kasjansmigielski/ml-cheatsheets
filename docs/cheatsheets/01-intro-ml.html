Wprowadzenie do Machine Learning ‚Äî fundament

ü§ñ Czym jest Machine Learning?

Machine Learning (ML) to dziedzina informatyki, kt√≥ra pozwala komputerom uczyƒá siƒô i podejmowaƒá decyzje na podstawie danych, bez konieczno≈õci programowania ka≈ºdej regu≈Çy z g√≥ry.

üí° Intuicja

Zamiast pisaƒá kod ‚Äúje≈õli temperatura > 25¬∞C, to bƒôdzie s≈Çonecznie‚Äù, ML pozwala algorytmowi samemu odkryƒá te zale≈ºno≈õci z historycznych danych pogodowych.

üìä G≈Ç√≥wne rodzaje ML

3) Reinforcement Learning (Uczenie ze wzmocnieniem)

Agent uczy siƒô przez interakcjƒô i nagrody/kary

# Przyk≈Çad koncepcyjny: optymalizacja reklam
class SimpleAgent:
    def __init__(self):
        # Jakie reklamy pokazywaƒá: ["sportowe", "technologiczne", "modowe"]
        self.ad_types = ["sportowe", "technologiczne", "modowe"]
        self.rewards = [0, 0, 0]  # nagrody za ka≈ºdy typ
        self.counts = [0, 0, 0]   # ile razy pokazane
    
    def choose_ad(self):
        # Wybierz reklamƒô z najwy≈ºszƒÖ ≈õredniƒÖ nagrodƒÖ
        avg_rewards = [r/max(c,1) for r, c in zip(self.rewards, self.counts)]
        return avg_rewards.index(max(avg_rewards))
    
    def update_reward(self, ad_type, clicked):
        # Aktualizuj nagrody na podstawie klikniƒôƒá
        self.counts[ad_type] += 1
        if clicked:
            self.rewards[ad_type] += 1
    
    def get_stats(self):
        for i, ad_type in enumerate(self.ad_types):
            rate = self.rewards[i] / max(self.counts[i], 1) * 100
            print(f"{ad_type}: {self.counts[i]} pokaza≈Ñ, {self.rewards[i]} klikniƒôƒá ({rate:.1f}%)")

# Symulacja
import numpy as np
np.random.seed(42)

agent = SimpleAgent()
print("üéØ Symulacja optymalizacji reklam:")
print("Agent uczy siƒô, kt√≥re reklamy dzia≈ÇajƒÖ najlepiej...\n")

for day in range(10):
    ad = agent.choose_ad()
    # R√≥≈ºne prawdopodobie≈Ñstwa klikniƒôƒá dla r√≥≈ºnych typ√≥w reklam
    click_probs = [0.1, 0.3, 0.2]  # technologiczne najlepsze
    clicked = np.random.random() < click_probs[ad]
    agent.update_reward(ad, clicked)
    print(f"Dzie≈Ñ {day+1}: pokazano {agent.ad_types[ad]}, klikniƒôta: {clicked}")

print("\nüìä Ko≈Ñcowe statystyki:")
agent.get_stats()

# Kompletny workflow ML na przyk≈Çadzie klasyfikacji iris
import pandas as pd
import numpy as np
from sklearn.datasets import load_iris
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score, classification_report

print("üî¨ Kompletny workflow projektu ML")
print("=" * 40)

# 1. Za≈Çaduj i poznaj dane
iris = load_iris()
data = pd.DataFrame(iris.data, columns=iris.feature_names)
data['target'] = iris.target
target_names = iris.target_names

print("1Ô∏è‚É£ Dane za≈Çadowane:")
print(f"Kszta≈Çt: {data.shape}")
print(f"Klasy: {target_names}")
print(data.head(3))

# 2. Przygotuj dane (czyszczenie, encoding)
X = data.drop('target', axis=1)
y = data['target']

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)
print(f"\n2Ô∏è‚É£ Dane przeskalowane (pierwsze 3 cechy pierwszej pr√≥bki):")
print(f"Przed: {X.iloc[0, :3].values}")
print(f"Po: {X_scaled[0, :3]}")

# 3. Podziel na train/test
X_train, X_test, y_train, y_test = train_test_split(X_scaled, y, test_size=0.2, random_state=42)
print(f"\n3Ô∏è‚É£ Podzia≈Ç danych:")
print(f"Train: {len(X_train)} pr√≥bek")
print(f"Test: {len(X_test)} pr√≥bek")

# 4. Wybierz i wytrenuj model
model = RandomForestClassifier(random_state=42)
model.fit(X_train, y_train)
print(f"\n4Ô∏è‚É£ Model wytrenowany: RandomForestClassifier")

# 5. Oce≈Ñ wyniki
predictions = model.predict(X_test)
accuracy = accuracy_score(y_test, predictions)
print(f"\n5Ô∏è‚É£ Wyniki:")
print(f"Dok≈Çadno≈õƒá: {accuracy:.2%}")

# Przyk≈Çad predykcji
sample_flower = X_test[0:1]
predicted_class = model.predict(sample_flower)[0]
predicted_name = target_names[predicted_class]
actual_name = target_names[y_test.iloc[0]]
print(f"\nPrzyk≈Çad predykcji:")
print(f"Przewidywana klasa: {predicted_name}")
print(f"Rzeczywista klasa: {actual_name}")
print("‚úÖ Poprawnie!" if predicted_name == actual_name else "‚ùå B≈ÇƒÖd")

actual_name = target_names[y_test.iloc[0]]
print(f"\nPrzyk≈Çad predykcji:")
print(f"Przewidywana klasa: {predicted_name}")
print(f"Rzeczywista klasa: {actual_name}")
print("‚úÖ Poprawnie!" if predicted_name == actual_name else "‚ùå B≈ÇƒÖd")
